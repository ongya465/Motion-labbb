// 🔥 추천 알고리즘
function recommend(){
  let part = document.getElementById("part").value;
  let exercise = rehabData[part];

  document.getElementById("result").innerText =
    "추천 운동: " + exercise;
}

// 🔥 평가 시스템
function evaluate(){
  let reps = Number(document.getElementById("reps").value);
  let pain = Number(document.getElementById("pain").value);
  let accuracy = Number(document.getElementById("accuracy").value);

  let score = (reps * 2) + (accuracy * 5) - (pain * 3);

  let text = "";
  let cls = "";

  if(score >= 80){
    text = "🔥 좋음";
    cls = "good";
  } else if(score >= 50){
    text = "👍 보통";
    cls = "mid";
  } else {
    text = "⚠️ 나쁨";
    cls = "bad";
  }

  document.getElementById("score").innerHTML =
    "점수: " + score + "<br>" + text;

  // 🤖 피드백
  let feedback = "";
  if(pain > 7) feedback = "통증 높음 → 강도 줄이기";
  else if(accuracy < 5) feedback = "자세 부정확 → 천천히";
  else feedback = "좋은 수행 👍";

  document.getElementById("feedback").innerText = feedback;
}
